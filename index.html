// Install dependencies: express const express = require('express'); const app = express(); // Initialize view count let viewCount = 0; // Serve static files (like HTML) app.use(express.static('public')); // API endpoint to get the live view count app.get('/api/get-view-count', (req, res) => { res.json({ viewCount }); }); // Increment view count each time a page is loaded app.get('*', (req, res) => { viewCount++; res.sendFile(__dirname + '/public/index.html'); }); // Start server const port = 3000; app.listen(port, () => { console.log(`Server is running on http://localhost:${port}`); });
